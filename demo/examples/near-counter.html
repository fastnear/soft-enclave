Contract Code

The counter contract uses NEAR storage API shim:

// Contract state stored via near.storageWrite()
function increment() {
  const count = near.storageRead('count') || 0;
  const newCount = count + 1;
  near.storageWrite('count', newCount);
  near.log('Incremented to ' + newCount);
  return newCount;
}

function decrement() {
  const count = near.storageRead('count') || 0;
  const newCount = count - 1;
  near.storageWrite('count', newCount);
  near.log('Decremented to ' + newCount);
  return newCount;
}

function getCount() {
  return near.storageRead('count') || 0;
}
